
==============================
Exercise 1: Basic Git Workflow
==============================
mkdir git-practice
cd git-practice
git init


touch README.md
git status
git add README.md
git commit -m "Initial commit with README"

echo "This is my Git practice repository" >> README.md
git add README.md
git commit -m "Update README file"

git log

==============================
Exercise 2: Branching and Merging
==============================
git branch feature
git checkout feature

touch feature.txt
echo "Feature branch work" > feature.txt
git add feature.txt
git commit -m "Add feature file"

git checkout main
git merge feature
git branch -d feature

==============================
Exercise 3: Conflict Resolution
==============================
git checkout -b branch1
echo "Line from branch1" > conflict.txt
git add conflict.txt
git commit -m "Branch1 change"

git checkout main
git checkout -b branch2
echo "Line from branch2" > conflict.txt
git add conflict.txt
git commit -m "Branch2 change"

git checkout main
git merge branch1
git merge branch2
# Resolve conflict manually, then:
git add conflict.txt
git commit -m "Resolve merge conflict"
